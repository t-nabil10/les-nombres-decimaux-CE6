الجمع والطرح - addition et soustraction

أ) تعريف:

  • الأعداد التي يتم جمعها أو طرحها تسمى حدوداً، وكل عدد يسمى حَداً
  • نتيجة الجمع تسمى المجموع
  • نتيجة الطرح تسمى الفرق

ب) مثال:

أحسب ما يلي:
800,80 + 560,7

الحل:

800,80 + 560,70 = 1361,50

الخطوات:

800,80 ← 800.80

560,70 ← 560.70

800.80 + 560.70 = 1361.50

587 - 93,54

الحل:

587,00 - 93,54 = 493,46

الخطوات:

587,00 ← 587.00

93,54 ← 93.54

587.00 - 93.54 = 493.46
